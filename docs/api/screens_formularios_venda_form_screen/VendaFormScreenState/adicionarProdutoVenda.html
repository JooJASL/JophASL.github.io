void adicionarProdutoVenda() {
  if (_produtoController.text.isEmpty) {
    focusProduto.requestFocus();
  } else if (_qtdProdutoController.text.isEmpty) {
    focusQtdProduto.requestFocus();
  } else if (_precoProdutoController.text.isEmpty) {
    focusPrecoProduto.requestFocus();
  } else if (empresaSelecionada == null || empresaSelecionada!.isEmpty) {
    ScaffoldMessenger.of(context)
        .showSnackBar(const SnackBar(content: Text("Selecione a empresa")));
  } else {
    setState(() {
      var produto = produtos![produtos!
          .indexWhere((element) => element['Id'] == _produtoSelecionado)];
      String? preco = _precoProdutoController.text;
      if (double.parse(_precoProdutoController.text) <
              double.parse(produto['Preco1']) &&
          usuario.venderAbaixoTabela == "NÃ£o") {
        preco = produto['Preco1'];
      }

      var produtoAdicionado = produtosVenda.indexWhere((produto) =>
          produto['idProduto'] == _produtoSelecionado &&
          produto['preco'] == preco);
      if (produtoAdicionado >= 0) {
        var qtdProduto =
            int.parse(produtosVenda[produtoAdicionado]['quantidade']) +
                int.parse(_qtdProdutoController.text);
        produtosVenda[produtoAdicionado]['quantidade'] =
            qtdProduto.toString();
        produtosVenda[produtoAdicionado]['quantidadeLoja'] =
            _qtdProdutoLojaController.text;
      } else {
        String? cfopProduto = "";
        compararEstados(produto).then((cfop) => cfopProduto = cfop);
        produtosVenda.add({
          "idProduto": _produtoSelecionado,
          "produto": _produtoController.text,
          "quantidade": _qtdProdutoController.text,
          "quantidadeLoja": _qtdProdutoLojaController.text.trim().isEmpty
              ? '0'
              : _qtdProdutoLojaController.text,
          "preco": preco,
          "cfop": cfopProduto != "" ? cfopProduto : _cfopSelecionado,
          "numeroCfop": listaCfops![listaCfops!
                  .indexWhere((element) => element['Id'] == _cfopSelecionado)]
              ['Cfop']
        });
      }
      totalCompra +=
          double.parse(_qtdProdutoController.text) * double.parse(preco!);
      _produtoController.text = "";
      _qtdProdutoController.text = "";
      _qtdProdutoLojaController.text = "";
      _precoProdutoController.text = "";
    });
  }
}
