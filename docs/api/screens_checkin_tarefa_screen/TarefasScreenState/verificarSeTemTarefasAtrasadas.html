List verificarSeTemTarefasAtrasadas(List? tarefas) {
  if (tarefas == null) return List.empty();

  final tarefasAtrasadas = [];
  for (final tarefa in tarefas) {
    final dataDoFim = DateTime.parse(
      "${tarefa['Data_Fim']} ${tarefa['Hora_Fim']}",
    );

    if (dataDoFim.isBefore(DateTime.now())) {
      tarefasAtrasadas.add(tarefa);
    }
  }

  return tarefasAtrasadas;
}
