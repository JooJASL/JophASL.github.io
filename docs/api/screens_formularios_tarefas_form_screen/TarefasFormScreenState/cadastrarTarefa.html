Future cadastrarTarefa() async {
  if (_formKey.currentState!.validate() == false) return;

  setState(() => enviandoFormulario.value = true);

  final tarefa = Tarefa(
    id: "",
    cliente: _clienteSelecionado ?? "",
    idUsuario: usuario.id ?? "",
    idResponsavel: usuario.idFuncionario ?? "",
    titulo: _tituloController.text,
    idUnidade: idUnidade,
    codigo: api.clientCode,
    observacao: _descricaoController.text,
    horaInicio: horaInicio!,
    horaFim: horaFinal!,
    dataInicio: dataInicio!,
    dataFim: dataFinal!,
  );

  final statusCode =
      await context.read<TarefasController>().cadastrarTarefa(tarefa);

  setState(() => enviandoFormulario.value = false);

  if (!mounted) return;
  switch (statusCode) {
    case 200:
      Navigator.pop(context);
      break;
    case 401:
      ScaffoldMessenger.of(context).showSnackBar(
          const SnackBar(content: Text('Fa√ßa login novamente')));
      break;
    case 500:
      ScaffoldMessenger.of(context).showSnackBar(
          const SnackBar(content: Text('Problema no servidor')));
      break;
  }
}
