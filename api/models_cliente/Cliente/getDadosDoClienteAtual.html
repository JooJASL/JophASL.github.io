Retorna uma lista de todos os clientes que o cliente logado atualmente possui.

static Future<List<Cliente>> getDadosDoClienteAtual() async {
  final usuario = infoUsuarioSingleton;
  final request = await http.get(
    Uri.parse(
        "${api.baseUrl}/clientes&codigoEmpresa=${api.clientCode}&idVendedor=${usuario.idFuncionario}"),
    headers: <String, String>{'CustomAuthorization': usuario.token!},
  );
  if (request.statusCode != 200) return [];

  final clientesJson = jsonDecode(request.body)['clientes'];
  final listaDeClientes = <Cliente>[];
  for (final cliente in clientesJson) {
    listaDeClientes.add(Cliente.fromJson(cliente));
  }

  return listaDeClientes;
}
