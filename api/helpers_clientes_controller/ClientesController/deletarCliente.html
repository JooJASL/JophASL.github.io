Future<http.Response> deletarCliente(Cliente cliente) async {
  http.Response response = await http.delete(
    Uri.parse(
        "${api.baseUrl}/clientes/excluir&cliente=${cliente.id}&codigoEmpresa=${api.clientCode}&idUsuario=${infoUsuarioSingleton.id}"),
    headers: <String, String>{
      'CustomAuthorization': infoUsuarioSingleton.token!
    },
  );

  if (response.statusCode != 200) return response;

  final responseJson = jsonDecode(response.body);
  if (responseJson['status'] == "ok") {
    _removerCliente(cliente);
    notifyListeners();
  }

  return response;
}
