Future<void> atualizarTarefa() async {
  enviandoFormulario.value = true;

  final tarefa = Tarefa(
    id: widget.tarefa!.id,
    idUnidade: idUnidade,
    cliente: _clienteSelecionado!,
    idUsuario: usuario.id!,
    idResponsavel: usuario.idFuncionario!,
    titulo: _tituloController.text,
    observacao: _descricaoController.text,
    dataInicio: dataInicio!,
    dataFim: dataFinal!,
    horaInicio: horaInicio!,
    horaFim: horaFinal!,
    status: status!,
    codigo: api.clientCode,
  );
  final response = await context.read<TarefasController>().atualizarTarefa(
        tarefa,
        jsonEncode(respostas),
        jsonEncode(notificar[0]),
      );

  enviandoFormulario.value = false;

  if (!mounted) return;

  switch (response.statusCode) {
    case 200:
      var responseJson = jsonDecode(response.body);
      if (responseJson['status'] == "ok") {
        var respostasCadastradas = responseJson['respostasCadastradas'];
        for (var i = 0; i < respostasCadastradas.length; i++) {
          var indexRetornoNull = respostas.indexWhere((e) => e.id == null);
          setState(() {
            respostas[indexRetornoNull].id = respostasCadastradas[i];
          });
        }
      }
      ScaffoldMessenger.of(context).showSnackBar(
          const SnackBar(content: Text("Atualização salva com sucesso")));
      break;
    case 401:
      ScaffoldMessenger.of(context).showSnackBar(
          const SnackBar(content: Text('Faça login novamente')));
      break;
    case 500:
      ScaffoldMessenger.of(context).showSnackBar(
          const SnackBar(content: Text('Problema no servidor')));
      break;
  }
}
